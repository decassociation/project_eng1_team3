325 		float fromBottomy = gameResolutionY - fromTopy;
326 		float x = Gdx.input.getX();
327 		boolean change = musSlide.getY() <= fromBottomy & fromBottomy <= musSlide.getY() + musSlide.getHeight();
328 		if (Gdx.input.isTouched() & change) {
329 			if (x >= musSlideBackgr.getX() & x <= musSlideBackgr.getX() + musSlideBackgr.getWidth()) {
330 				musSlide.setPosition(Gdx.input.getX(), musSlide.getY());
331 				v = (musSlide.getX() - musSlideBackgr.getX()) / musSlideBackgr.getWidth();
