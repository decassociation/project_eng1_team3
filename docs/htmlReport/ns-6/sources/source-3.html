833 	private void constructCollisionData(TiledMap mp) {
834 		TiledMapTileLayer botlayer = (TiledMapTileLayer) mp.getLayers().get(0);
835 		int mapwidth = botlayer.getWidth();
836 		int mapheight = botlayer.getHeight();
837 		CLTiles = new CollisionTile[mapwidth][mapheight];

839 		int topwidth = toplayer.getWidth();
840 		int topheight = toplayer.getHeight();

843 				TiledMapTileLayer.Cell tl2 = toplayer.getCell(x, y);
844 				if (tl2 != null) {
845 					CLTiles[x][y] = new CollisionTile(x * 64, y * 64, 64, 64);

848 		}
849 		for (int y = 0; y < mapheight; y++) {
850 			for (int x = 0; x < mapwidth; x++) {
851 				TiledMapTileLayer.Cell tl = botlayer.getCell(x, y);
852 				if (tl != null) {
853 					TiledMapTile tlt = tl.getTile();
854 					MapProperties mpr = tlt.getProperties();
855                     //Checks if tile have the "name" custom property

857 					if (mpr.get("name") == null) {
858 						CLTiles[x][y] = new CollisionTile(x * 64, y * 64, 64, 64);
859 					} else {
860 						if (y != 0) {

862 								CLTiles[x][y - 1] = null;
